database = books_db

books table =>
    CREATE TABLE books (
        id INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
        title VARCHAR(100),
        name VARCHAR(100),
        size INT(4),
        mime VARCHAR(50),
        path text,
        created DATETIME DEFAULT NULL,
        modified DATETIME DEFAULT NULL,
        user_id INT(11) NOT NULL
    );

    

users table =>
    CREATE TABLE users (
        name VARCHAR(100) NOT NULL,
        id INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
        username VARCHAR(50),
        password VARCHAR(255),
        role VARCHAR(20),
        created DATETIME DEFAULT NULL,
        modified DATETIME DEFAULT NULL 
    );
